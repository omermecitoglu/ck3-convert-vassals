convert_vassals_toggle_converting_subvassals = {
	scope = character
	is_shown = {
		has_global_variable = ignore_converting_subvassals
	}
	effect = {
		if = {
			limit = {
				has_global_variable = ignore_converting_subvassals
			}
			remove_global_variable = ignore_converting_subvassals
		}
		else = {
			set_global_variable = ignore_converting_subvassals
		}
	}
}
